making api request from the components
two ways: 1.fetch
            ex:get request- let res = await fetch("url",{method:"GET"}) // if we dont specify the 'method', then default is get
                            let data = await res.json() // this data variable now contains the content, {message:"",payload:""},extraction is done manually by calling the json method
                            if(data.status !== 200){//throw some error
                                throw new Error("some error")
                            }
                            //if any request is made, the application gets to know whether that request was successful or not by checking the status, else we have to manually create some error handling logic
            ex: post request - 
                let resObj = await fetch("url",{method:"POST",headers:{"Content-Type":"application/json"},body:JSON.stringify()})
                //we need to manually set the header to content-type:application/json and we must convert the js object to json using JSON.stringify()
          2. AXIOS
                it is not an inbuilt feature in js, we have to install it and import it
                ex: let resObj = await axios.get("") // on the axios we can specify what kind of request it is
                    let res = resObj.data; // here there is no need to manually extract, the while the request is made only the data will get extracted
                    //if there are any errors then the axios automatically sends the error object......
                ex: let resObj = await axios.post("url",obj) // this object automatically gets the data object from the body
                    let res = resObj.data
                ->this is less verbose, simpler and has internal error handling
                ->it contains err.response.<!--something-->
